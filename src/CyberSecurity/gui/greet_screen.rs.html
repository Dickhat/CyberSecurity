8#[derive(Debug, Clone)]
9pub struct Credentials {
10    pub login: String,
11    password: String,
12    pub access: bool,
13    info_message: String,
14}
15
16#[derive(Debug, Clone)]
17pub enum Message {
18    Login(Credentials),
19    Registration(Credentials),
20    RegistrationResult(bool),
21    AnonAccess,
22    Authorize(bool),
23    Filling(Credentials),
24    Error(String),
25}
26
27impl Credentials {
28    pub fn new() -> Self {
29        Self {
30            login: "".to_string(),
31            password: "".to_string(),
32            access: false,
33            info_message: "".to_string(),
34        }
35    }
36
37    // Обновляет состояние
38    pub fn update(&mut self, message: Message) -> iced::Task<Message> {
39        match message {
40            Message::Filling(credentials) => {
41                self.login = credentials.login;
42                self.password = credentials.password;
43            }
44            Message::Login(credentials) => {
45                return Task::perform(async move { auth_credential(credentials) }, |result| {
46                    match result {
47                        Ok(auth_status) => {
48                            return Message::Authorize(auth_status);
49                        }
50                        Err(message_err) => {
51                            return Message::Error(message_err);
52                        }
53                    };
54                })
55            }
56            Message::Registration(credentials) => {
57                return Task::perform(async move { reg_credential(credentials) }, |result| {
58                    Message::RegistrationResult(result)
59                })
60            }
61            Message::AnonAccess => {
62                self.login = String::from("Anon");
63                self.access = true;
64                return Task::done(Message::Authorize(true));
65            }
66            Message::RegistrationResult(success) => {
67                if success {
68                    self.info_message =
69                        "Пользователь успешно зарегистрирован. Авторизуйте по вашим данным."
70                            .to_string();
71                } else {
72                    self.info_message =
73                        "Пользователь не зарегистрирован. Попробуйте снова".to_string();
74                }
75            }
76            Message::Authorize(access) => {
77                self.access = access;
78
79                if !access {
80                    self.info_message = "Некорректные данные".to_string();
81                }
82            }
83            Message::Error(err_message) => {
84                self.info_message = err_message;
85            }
86        }
87
88        iced::Task::none()
89    }

123            button("Register")

125                .padding([10, 120])

142fn check_exist_user(login: &String, hash_password: &String) -> bool {
143    // Connect to the database.
144    let mut client = Client::connect(
145        "host=localhost user=postgres password=postgres dbname=CyberSecurity",
146        NoTls,
147    )
148    .unwrap();
149
150    // Запрос существования пользователя с текущими данными
151    let row = client
152        .query_one(
153            "SELECT COUNT(*) FROM public.users WHERE login = $1 AND password_hash = $2",
154            &[&login, &hash_password],
155        )
156        .unwrap();
157
158    let count: i64 = row.get(0);
159
160    if count != 0 {
161        return true;
162    } else {
163        return false;
164    }
165}

167/// Проверка, что пользователь ввел данные, существующие в БД
168fn auth_credential(credentials: Credentials) -> Result<bool, String> {
169    if credentials.login == "" || credentials.password == "" {
170        return Err("Логин или пароль не введены".to_string());
171    }
172
173    // Перевод в байты
174    let pass_vec = credentials.password.as_bytes();
175
176    // Получение Хэша пароля
177    let hash_vec = streebog(&pass_vec[..], 512)?;
178    let hash = to_hex(&hash_vec);
179
180    Ok(check_exist_user(&credentials.login, &hash))
181}

183/// Попытка зарегистрировать нового пользователя
184fn reg_credential(credentials: Credentials) -> bool {
185    if credentials.login == "" || credentials.password == "" {
186        return false;
187    }
188
189    // Перевод в байты
190    let pass_vec = credentials.password.as_bytes();
191
192    // Получение Хэша пароля
193    let hash_vec = match streebog(&pass_vec[..], 512) {
194        Ok(hash) => hash,
195        Err(_) => {
196            return false;
197        }
198    };
199
200    let hash = to_hex(&hash_vec);
201
202    if check_exist_user(&credentials.login, &hash) {
203        return false;
204    }
205
206    // Connect to the database.
207    let mut client = Client::connect(
208        "host=localhost user=postgres password=postgres dbname=CyberSecurity",
209        NoTls,
210    )
211    .unwrap();
212
213    match client.query_opt(
214        "INSERT INTO public.users (login, password_hash) VALUES($1, $2)",
215        &[&credentials.login, &hash],
216    ) {
217        Ok(_) => true,
218        Err(_) => false,
219    }
