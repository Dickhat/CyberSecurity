6#[derive(Debug, Clone)]
7struct Credentials {
8    login: String,
9    password: String,
10    access: bool,
11    info_message: String
12}
13
14#[derive(Debug, Clone)]
15enum Message {
16    Login(Credentials),
17    Registration(Credentials),
18    RegistrationResult(bool),
19    AnonAccess,
20    Authorize(bool),
21    Filling(Credentials),
22    Error(String)
23}
24
25impl Credentials 
26{
27    fn new() -> Self {
28        Self { login: "".to_string(), password: "".to_string(), access: false, info_message: "".to_string()}
29    }
30
31    // Обновляет состояние 
32    fn update(&mut self, message: Message) -> iced::Task<Message>
33    {
34        match message {
35            Message::Filling(credentials) => {self.login = credentials.login; self.password = credentials.password;},
36            Message::Login(credentials) => return Task::perform(async move {auth_credential(credentials)},
37                |result| {
38                    match result {
39                        Ok(auth_status) => {return Message::Authorize(auth_status);},
40                        Err(message_err) => {return Message::Error(message_err);}
41                    };
42            }),
43            Message::Registration(credentials) => return Task::perform(
44                async move {reg_credential(credentials)}, |result| {Message::RegistrationResult(result)}),
45            Message::AnonAccess => {self.login = String::from("Anon"); self.access = true;},
46            Message::RegistrationResult(success) => {
47                if success{self.info_message = "Пользователь успешно зарегистрирован. Авторизуйте по вашим данным.".to_string();}
48                else {self.info_message = "Пользователь не зарегистрирован. Попробуйте снова".to_string();}
49            },
50            Message::Authorize(access) => {
51                self.access = access;
52
53                if !access {self.info_message = "Некорректные данные".to_string();}
54            },
55            Message::Error(err_message) => {self.info_message = err_message;}
56        }
57
58        iced::Task::none()
59    }
60
61    // Обновляет UI
62    fn view(&self) -> iced::Element<'_, Message>
63    {
64        if self.access 
65        {
66            iced::widget::text(format!("User: {}", self.login)).into()
67        }
68        // Введение Логина и Пароля
69        else {
70            let greet_form = column![
71                text(self.info_message.clone()),
72                text_input("Login", &self.login).on_input({
73                    move |login| Message::Filling(Credentials {
74                        login,
75                        password: self.password.clone(),
76                        access: false,
77                        info_message: "".to_string()
78                    })
79                })
80                    .width(500),
81                text_input("Password", &self.password).on_input({
82                    move |password| Message::Filling(Credentials {
83                        login: self.login.clone(),
84                        password,
85                        access: false,
86                        info_message: "".to_string()
87                    })
88                })
89                    .width(500),
90                button("Log in")
91                    .width(300)
92                    .padding([10, 120])
93                    .on_press(Message::Login(self.clone())),
94                button("Register")
95                    .width(300)
96                    .padding([10, 120])
97                    .on_press(Message::Registration(self.clone())),
98                button("Anon access")
99                    .width(300)
100                    .padding([10, 100])
101                    .on_press(Message::AnonAccess)
102            ].align_x(iced::Alignment::Center);//.into()
103
104            container(greet_form)
105                                            .padding(100)
106                                            .center(1600)
107                                            .style(container::rounded_box)
108                                            .into()
109        }
110    }
111}

113fn check_exist_user(login: &String, hash_password: &String) -> bool
114{
115    // Connect to the database.
116    let mut client =
117        Client::connect("host=localhost user=postgres password=postgres dbname=CyberSecurity" , NoTls).unwrap();
118
119    // Запрос существования пользователя с текущими данными
120    let row = client
121        .query_one("SELECT COUNT(*) FROM public.users WHERE login = $1 AND password_hash = $2", &[&login, &hash_password])
122        .unwrap();
123
124    let count: i64 = row.get(0);
125
126    if count != 0 {return true;} else {return false;}
127}

129/// Проверка, что пользователь ввел данные, существующие в БД
130fn auth_credential(credentials:Credentials) -> Result<bool, String>
131{
132    if credentials.login == "" || credentials.password == "" {return Err("Логин или пароль не введены".to_string());}
133
134    // Перевод в байты
135    let pass_vec = credentials.password.as_bytes();
136
137    // Получение Хэша пароля
138    let hash_vec = streebog(&pass_vec[..], 512)?;
139    let hash = to_hex(&hash_vec);
140    
141    Ok(check_exist_user(&credentials.login, &hash))
142}
143
144/// Попытка зарегистрировать нового пользователя
145fn reg_credential(credentials:Credentials) -> bool
146{
147    if credentials.login == "" || credentials.password == "" {return false;}
148
149    // Перевод в байты
150    let pass_vec = credentials.password.as_bytes();
151
152    // Получение Хэша пароля
153    let hash_vec = match streebog(&pass_vec[..], 512)
154    {
155        Ok(hash) => hash,
156        Err(_) => {return false;}
157    };
158
159    let hash = to_hex(&hash_vec);
160    
161    if check_exist_user(&credentials.login, &hash) {return false;}
162
163    // Connect to the database.
164    let mut client =
165        Client::connect("host=localhost user=postgres password=postgres dbname=CyberSecurity" , NoTls).unwrap();
166
167    match client.query_opt("INSERT INTO public.users (login, password_hash) VALUES($1, $2)", &[&credentials.login, &hash])
168    {
169        Ok(_) => true,
170        Err(_) => false
171    }
172}
