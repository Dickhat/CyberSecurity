18    fn modpow(mut base: U512, mut exp: U512, modulus: U512) -> U512 {
19        if modulus == U512::ONE {
20            return U512::ZERO;
21        }
22
23        let modulus_nz: NonZero<U512> = NonZero::new(modulus).expect("modulus must be nonzero");
24        let mut result: U512 = U512::ONE;
25        base = base % modulus;
26
27        while exp > U512::ZERO {
28            if exp.is_odd().into() {
29                result = result.mul_mod(&base, &modulus_nz);
30            }
31
32            exp = exp >> 1;
33            base = base.mul_mod(&base, &modulus_nz);
34        }
35        result
36    }

50    pub fn bpn_check(number: U256) -> bool {
51        let mut s: U256 = U256::ZERO;
52        let mut d: U256 = number - U256::ONE;
53
54        // Получаем s и d из выражения n - 1 = (2^s)*d
55        while d.is_even().into() {
56            s += U256::ONE;
57            d = d.shr(1);
58        }
59
60        // Расчет значения ⌊2(ln_number)^2⌋ в виде floor(pow(ln2/lnE,2))
61        // ln2
62        let log2_n = if number.bits() > 0 {
63            (number.bits() - 1) as f64
64        } else {
65            0 as f64
66        };
67
68        let ln_n = log2_n / std::f64::consts::LOG2_E;
69        let floor_ln_n = U256::from((2.0 * ln_n.powi(2)).floor() as u128);
70
71        let min: U256 = if number - U256::ONE - U256::ONE < floor_ln_n {
72            number - U256::ONE - U256::ONE
73        } else {
74            floor_ln_n
75        };
76
77        let mut x: U256;
78        let mut y: U256;
79
80        let mut a: U256 = U256::from(2u32);
81
82        // for all a in the range [2, min(n − 2, ⌊2(ln_number)^2⌋)]:
83        while a < min {
84            // x ← a^d mod n
85            // x = Self::modpow(a, d, number);
86
87            // Долго и вылетает с переполнением
88            let mut i: U256 = U256::ONE;
89            let mut temp_x: U256 = a % number;
90
91            // Возведение в степень d - 1 раз
92            while i < d {
93                temp_x = temp_x * a % number;
94
95                i += U256::ONE;
96            }
97
98            x = temp_x;
99
100            let mut i: U256 = U256::ZERO;
101
102            y = x;
103
104            while i < s {
105                // y = x^2 mod number;
106                y = x % number;
107                y = y * x % number;
108
109                if y == U256::ONE && x != U256::ONE && x != (number - U256::ONE) {
110                    return false;
111                }
112
113                x = y;
114
115                i += U256::ONE;
116            }
117
118            // Уточнить про ситуацию неинициализированного y при s = 0
119            if y != U256::ONE {
120                return false;
121            }
122
123            a += U256::ONE
124        }
125

331        let a = *left % *modulus;
332        let b = *rigth % *modulus;
333
334        let mut result = U32768::ZERO;
335        let mut base = a;
336
337        // Пробегаем по всем 32768 битам b
338        for i in 0..32768 {
339            if b.bit(i).into() {
340                if result.saturating_add(&base) == U32768::MAX
341                {
342                    // Перенос старшей части в младщую
343                    let elder_bits_result = result.shr(16384);
344                    let elder_bits_base = base.shr(16384);
345
346                    // Суммирование с модулем, которые тоже перенесен в младшую часть
347                    let mut overflow_result = (elder_bits_base + elder_bits_result) % (*modulus).shr(16384);
348
349                    // Возврат результата в старшую часть
350                    overflow_result = overflow_result.shl(16384);
351
352                    // Суммирование младшей части с взятием модуля
353                    let lower_bits = (result.shl(16384).shr(16384) + base.shl(16384).shr(16384)) % *modulus;
354
355                    // Суммирование старшей и младшей частей с взятием модуля
356                    result = (overflow_result + lower_bits) % *modulus;
357                }
358                else {
359                    result = (result + base) % *modulus;
360                }
361            }
362
363            // Удваиваем base (умножаем на 2) и делаем модуль сразу
364            base = base.shl(1) % *modulus;
365        }
366
367        result
