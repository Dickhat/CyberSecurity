26    fn modpow(mut base: U512, mut exp: U512, modulus: U512) -> U512 {
27        if modulus == U512::ONE
28        {
29            return U512::ZERO;
30        }
31
32        let modulus_nz: NonZero<U512> = NonZero::new(modulus).expect("modulus must be nonzero");
33        let mut result: U512 = U512::ONE;
34        base = base % modulus;
35
36        while exp > U512::ZERO {
37            if exp.is_odd().into() {
38                result = result.mul_mod(&base, &modulus_nz);
39            }
40            
41            exp = exp >> 1;
42            base = base.mul_mod(&base, &modulus_nz);
43        }
44        result
45    }

60    fn bpn_check(number:U256) -> bool
61    {
62        let mut s:U256 =  U256::ZERO;
63        let mut d:U256 = number - U256::ONE;
64
65        // Получаем s и d из выражения n - 1 = (2^s)*d
66        while d.is_even().into()
67        {
68            s += U256::ONE;
69            d = d.shr(1);
70        }
71
72        // Расчет значения ⌊2(ln_number)^2⌋ в виде floor(pow(ln2/lnE,2))
73        // ln2
74        let log2_n  = if number.bits() > 0 {
75            (number.bits() - 1) as f64
76        } else {
77            0 as f64
78        };
79
80        let ln_n = log2_n / std::f64::consts::LOG2_E;
81        let floor_ln_n = U256::from((2.0*ln_n.powi(2)).floor() as u128);
82
83        let min:U256 = if number - U256::ONE - U256::ONE < floor_ln_n
84        {
85            number - U256::ONE - U256::ONE
86        }
87        else {
88            floor_ln_n
89        };
90
91        let mut x:U256;
92        let mut y:U256;
93
94        let mut a:U256 = U256::from(2u32);
95        
96        // for all a in the range [2, min(n − 2, ⌊2(ln_number)^2⌋)]:
97        while a < min
98        {
99            // x ← a^d mod n
100            // x = Self::modpow(a, d, number);
101
102            // Долго и вылетает с переполнением
103            let mut i:U256 = U256::ONE;
104            let mut temp_x:U256 = a % number;
105
106            // Возведение в степень d - 1 раз
107            while i < d
108            {
109                temp_x = temp_x * a % number;
110
111                i += U256::ONE;
112            }
113
114            x = temp_x;
115
116            let mut i:U256 = U256::ZERO;
117
118            y = x;
119
120            while i < s
121            {
122                // y = x^2 mod number;
123                y = x % number;
124                y = y * x % number;
125
126                if y == U256::ONE && x != U256::ONE && x != (number - U256::ONE) {return false;}
127                
128                x = y;
129
130                i += U256::ONE;
131            }
132
133            // Уточнить про ситуацию неинициализированного y при s = 0
134            if y != U256::ONE {return false;}
135
136            a += U256::ONE
137        }
138
139        return true;
140    }

239        while e > phi_n || !Self::coprime(phi_n, e)

256    /// Шифрование с дополнением. Старшие 2 байта нулевые, чтобы m < N. Старший 3 байт обозначает 
257    /// число байт дополнения сообщения до 64 байт.
258    /// Пример: 00A0000000BACSA...A <- старшие 00, A - число байт дополнения, начало сообщения с байта со значением B. 
259    pub fn encryption(&self, message:& String) -> Result<PathBuf, String>
260    {
261        let path = PathBuf::from("output_encryption");
262        let mut file = File::create(&path).unwrap();     // Добавить проверку на создание файла
263        let mut buf = [0u8; 64];
264        let mut cipher_text:U512;
265
266        // Короткий текст
267        if message.len() < 64
268        {
269            buf[2] = 64 - message.as_bytes().len() as u8; // Число байт дополнения (включая сам этот байт)
270            buf[64 - message.as_bytes().len()..].copy_from_slice(message.as_bytes()); // Вставить в конец сообщение
271
272            cipher_text = U512::from_be_slice(&buf);
273            cipher_text = Self::modpow(cipher_text, self.public_key, self.n);
274
275            file.write_all(&cipher_text.to_be_bytes()).unwrap();
276        }
277        
278        else {
279            let b_message = message.as_bytes();
280
281            // По 60 байт (text < (512 / 8))
282            for bytes in b_message.chunks(60)
283            {
284                buf = [0u8; 64];
285                buf[2] = 64 - bytes.len() as u8;                    // Число байт дополнения (включая сам этот байт)
286                buf[64 - bytes.len()..].copy_from_slice(bytes); // Вставить в конец сообщение
287
288                cipher_text = U512::from_be_slice(&buf);
289                cipher_text = Self::modpow(cipher_text, self.public_key, self.n);
290
291                // Добавить проверку на записанные байты
292                file.write_all(&cipher_text.to_be_bytes()).unwrap();
293            }
294        }
295
296        Ok(path)
297    }

299    /// Расшифрование происходит по обратной схеме шифрованию. 
300    /// Считывается число нулей, добавленных дополнением и они убираются из исходного сообщения.
301    pub fn decryption(&self, message_path:PathBuf) -> Result<PathBuf, String>
302    {
303        let path = PathBuf::from("output_decrytion");
304
305        // Добавить проверку на создание файла
306        let mut file_out = File::create(&path).unwrap();
307        let mut file_in = File::open(&message_path).unwrap();
308        
309        let mut buffer = [0u8; 64];
310        let mut text:U512;
311        let mut padding:usize;
312
313        loop {
314            let n = file_in.read(&mut buffer).unwrap();
315
316            if n == 0 {break;}
317
318            text = U512::from_be_slice(&buffer);
319            text = Self::modpow(text, self.private_key, self.n);
320
321            let text_b = text.to_be_bytes();
322            padding = 64 - text_b[2] as usize;       // ПРоверить переполнение
323
324            // Добавить проверку на записанные байты
325            file_out.write_all(&text_b[64 - padding..]).unwrap(); // Записать текст без дополнения
326        }
327
328        Ok(path)
329    }
